## Install

```bash
npm install --save vnxy
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'vnxy'
import 'vnxy/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
